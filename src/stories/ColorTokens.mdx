export const colors = Array.from(
  Object.entries(canonicalColorTokens)
    .filter(([, value]) => isColor(value))
    .reduce((acc, [name, value]) => {
      const key = value.toUpperCase();
      const current = acc.get(key);

      if (current) {
        current.names.push(name);
        return acc;
      }

      acc.set(key, {
        value,
        names: [name],
        meta: colorMeta[name],
      });

      return acc;
    }, new Map())
    .values(),
);
