# Иконки

Все иконки экспортируются из пакета и принимают одинаковые props.

```tsx
import { FilterIcon } from 'mathmech-ui-kit';

<FilterIcon size={24} color="currentColor" strokeWidth={2} />
```

<div className="icon-props">
  <span>size: 16 | 24 | 32 | 48 | 64</span>
  <span>color: CSS color string</span>
  <span>strokeWidth: 1 | 2</span>
</div>

<div className="icons-grid">
  {icons.map(({ name, component: Icon }) => (
    <div className="icon-card" key={name}>
      <div className="icon-preview">
        <Icon size={32} color="currentColor" strokeWidth={2} />
      </div>
      <code>{name}</code>
    </div>
  ))}
</div>
